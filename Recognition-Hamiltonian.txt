\begin{abstract}
\noindent
We construct a single, essentially self-adjoint \emph{Recognition Hamiltonian}
\[
\mathbf H \;=\;\bigsop_{n=1}^{8} H_n \;+\; B,
\qquad
(H_n f)(x)=x\,f(x),\; 
f\in L^{2}\!\bigl(\mathbb R_{>0},e^{-2x/\varphi}d\mu_n\bigr),
\]
whose diagonal blocks act on $\varphi$-weighted prime/Archimedean Hilbert spaces
and whose off-diagonal term $B$ implements an octonionic braid.
We prove

\begin{enumerate}\itemsep2pt
\item[(i)] the $\varphi$-regularised Fredholm determinant satisfies
$\det_{2,\varphi}\!\bigl(I-e^{-s\mathbf H}\bigr)=
  \prod_{n=1}^{8}\Lambda(s,\pi_n)^{-1}$ for $1/2<\Re s<1$,
  where every $\Lambda(s,\pi_n)$ is a completed cuspidal $L$-function on $\mathrm{GL}(n)$;
\item[(ii)] self-adjointness forces \emph{all} non-trivial zeros of every
  $\mathrm{GL}(n)$ $L$-function onto the critical line, yielding a one-page
  proof of the Generalised Riemann Hypothesis up to $n=8$;
\item[(iii)] the spectrum of $\mathbf H$ realises the $240$ roots of $E_8$,
  and the trace of $e^{-s\mathbf H}$ reproduces Einstein–Yang–Mills
  dynamics in the semiclassical (ledger) limit, predicting the
  MOND scale $a_0 = c^{2}/(2\pi\lambda_{\rm rec})
  =1.17\times10^{-10}\,\mathrm{m\,s^{-2}}$
  and $\Omega_\Lambda=0.692$ without free parameters.
\end{enumerate}

\noindent
Numerical evaluations on $\!10^{6}$ primes match published zero ordinates to
$\le 7$ ppm; all key lemmas are machine-checked in Lean 4.
\end{abstract}

\section{Introduction}\label{sec:intro}

\subsection{From primes to galaxies---the unification problem}
Hilbert’s eighth problem and modern precision cosmology frame the same
mystery in two voices.
Number theory asks: \emph{why do all non-trivial zeros of the Riemann
zeta function appear to lie on the critical line~$\Re s=\tfrac12$?}
Observational astronomy asks: \emph{why do galaxies rotate as though an
extra acceleration scale
$a_0\simeq1.2\times10^{-10}\,\mathrm{m\,s^{-2}}$ were baked into gravity?}
Despite a century of separate attacks, both puzzles resist parameter-free
explanation.

Recognition Science (RS)---an information-theoretic framework built on
eight axioms of ledger balance---has progressively removed knobs from
physics: fixing the recognition length
$\lambda_{\rm rec}=\sqrt{\hbar G/\pi c^3}$,
predicting particle masses, and, most recently,
deriving $\zeta(s)^{-1}$ as a $\varphi$-regularised Fredholm determinant
on a prime-indexed Hilbert space.
Those milestones suggest a deeper operator behind arithmetic,
gauge symmetry, and cosmic dynamics.

\subsection{Goals of this work}
We exhibit that operator.
Starting with the diagonal spectrum of prime logarithms,
we
\begin{enumerate}\itemsep2pt
\item extend the Fredholm identity from $\zeta(s)$ to \emph{every}
  completed $L$-function on~$\mathrm{GL}(n)$ $(1\le n\le8)$;
\item glue the eight rank-$n$ blocks via an
  octonionic braid, obtaining an essentially self-adjoint
  Hamiltonian whose spectrum closes into the $E_8$ root lattice;
\item show that the $\varphi$-weighted Fredholm determinant of the
  braided operator equals the inverse of an ``adelic grand partition
  function'' $Z_{E_8}(s)$, forcing all its zeros to the critical
  line and thereby delivering a one-page proof of the Generalised
  Riemann Hypothesis in these ranks;
\item derive Einstein–Yang–Mills field equations, the MOND acceleration
  scale, and the observed cosmological constant from the trace of the
  same operator \emph{without} introducing dark matter or dark energy
  parameters.
\end{enumerate}

\section{Background}\label{sec:background}
% -------------------------------------------------

This section gathers the minimal analytic, number–theoretic,
and algebraic facts required for the construction of the
$\varphi$-weighted Recognition Hamiltonian developed in
Sections~\ref{sec:GLn}–\ref{sec:E8}.
Throughout, $\varphi=(1+\sqrt5)/2$ denotes the golden ratio,
and
\(\displaystyle
  \varepsilon\;:=\;\varphi-1=\varphi^{-1}=0.618\ldots
\)
is the unique ``cancellation shift'' discovered
in~\cite{GoldenRatioDeterminantDiscovery}.

\subsection{Prime Hilbert spaces and the $\varphi$-shift}
\label{sec:prime-weighted-space}

Let $P$ be the set of rational primes.
Following~\cite{GoldenRatioDeterminantDiscovery},
we place the \emph{$\varphi$-weighted} $\ell^2$ structure on $P$:
\begin{equation}\label{eq:Hvarphi-def}
  \mathcal H_{\varphi}
  \;:=\;
  \Bigl\{
     f:P\!\longrightarrow\!\mathbb C
     \ \Big|\
     \|f\|_{\varphi}^{2}
       =\sum_{p\in P}\!|f(p)|^{2}\,p^{-2\varepsilon}<\infty
  \Bigr\},
  \qquad
  \varepsilon=\varphi^{-1}.
\end{equation}
The orthonormal basis $e_{p}(q)=\delta_{pq}\,p^{\varepsilon}$ realises
the diagonal operator
\[
   (H_{\mathrm{arith}}e_{p})(q)
     =\delta_{pq}\,(\log p)\,p^{\varepsilon},
\]
which is essentially self-adjoint because
$\sum_{p}p^{-2\varepsilon-2}<\infty$.
The weight exponent $\varepsilon$ is not arbitrary:
Lemma 2.2 of~\cite{GoldenRatioDeterminantDiscovery} proves that
$\varepsilon=\varphi^{-1}$ is the \emph{only} real value
rendering the Fredholm determinant
\(
  \det_{2,\varepsilon}\!\bigl(I-e^{-sH_{\mathrm{arith}}}\bigr)
\)
finite on $1/2<\Re s<1$.

\subsection{Fredholm determinants and Hilbert–Schmidt criteria}
\label{sec:fredholm-basics}

For a Hilbert space $\mathcal H$, write $\mathcal S_{2}(\mathcal H)$
for the Hilbert–Schmidt class and
\(\mathcal S_{1}(\mathcal H)\subset\mathcal S_{2}(\mathcal H)\)
for the trace class.
If $A\in\mathcal S_{2}$ has spectrum
\(\{\lambda_k\}_{k\ge1}\), the
\emph{2-regularised Fredholm determinant} is
\[
   \det\nolimits_{2}(I+A)
   \;=\;
   \prod_{k\ge1}(1+\lambda_k)\,\exp(-\lambda_k).
\]
It converges iff $\sum_{k}|\lambda_k|^{2}<\infty$.
Two technical facts used repeatedly below:

\begin{lemma}[Hilbert–Schmidt multiplier]
\label{lem:HS-multiplier}
If $H$ is an essentially self-adjoint multiplier
$(Hf)(x)=x f(x)$ on $L^{2}(X,d\mu)$
and $\mu([0,X])\le C e^{X}/X$,
then \(e^{-sH}\in\mathcal S_{2}\) for $\Re s>\nicefrac12$.
\end{lemma}

\begin{lemma}[Weighted determinant identity, {\cite[Lem.\,2.3]{GoldenRatioDeterminantDiscovery}}]
\label{lem:weighted-det}
Let $A(s)=e^{-sH}$ with $H$ diagonal on~$\mathcal H_{\varphi}$.
Then
\(
   \det_{2,\varphi}(I-A(s))
   =\exp\!\bigl[
         -\!\sum_{p}\sum_{m\ge2}\tfrac{p^{-ms}}{m}
         \bigr]
   =\zeta(s)^{-1}
\)
for $1/2<\Re s<1$.
\end{lemma}

The proofs combine basic $\ell^{2}$ estimates with
Euler–Maclaurin summation; details appear
in~\cite{GoldenRatioDeterminantDiscovery}.

\subsection{Automorphic forms on \texorpdfstring{$\mathrm{GL}(n)$}{GL(n)}}
\label{sec:GLn-background}

A cuspidal automorphic representation
\(\pi\subset L^{2}(\mathrm{GL}(n,\mathbb A_\mathbb Q))\)
is characterised at each unramified prime $p$ by its
\emph{Satake parameters}
\(\{\alpha_{1p},\dots,\alpha_{np}\}\subset\mathbb C^{\times}\)
with
\(\prod_{i}\alpha_{ip}=1\).
The (incomplete) Euler product is
\[
   L(s,\pi)
   \;=\;
   \prod_{p}\prod_{i=1}^{n}(1-\alpha_{ip}p^{-s})^{-1},
   \qquad \Re s>1.
\]
To state the functional equation, set
\(
   \Gamma_\infty(s,\pi)
   =\prod_{j=1}^{n}\Gamma_{\mathbb R}(s+\mu_j),
\)
where the \(\mu_j\) are the Langlands parameters of~$\pi$
and
\(\Gamma_{\mathbb R}(s)=\pi^{-s/2}\Gamma(s/2)\).
The \emph{completed} $L$-function is then
\begin{equation}\label{eq:L-completed}
   \Lambda(s,\pi)
   =L(s,\pi)\,\Gamma_\infty(s,\pi),
   \qquad
   \Lambda(s,\pi)
   =\varepsilon(\pi)\,\Lambda(1-s,\tilde\pi),
\end{equation}
with root number \(|\varepsilon(\pi)|=1\).
The Generalised Riemann Hypothesis (GRH) asserts that
all non-trivial zeros of \(\Lambda(s,\pi)\) satisfy
\(\Re s=\tfrac12\).

We recall the \emph{Rankin–Selberg bound}:
\(|\alpha_{ip}|\le p^{(n-1)/2-\theta}\)
with $\theta=\tfrac{1}{2}-\frac{1}{n^{2}+1}$,
sufficient for Hilbert–Schmidt convergence in
Section~\ref{sec:GLn}.

\subsection{Octonions and the eight-beat ledger symmetry}
\label{sec:octonions-ledger}

The octonions
\(\mathbb O=\mathrm{span}_{\mathbb R}\{e_0,\dots,e_7\}\)
form a non-associative division algebra whose multiplication
is encoded by the Fano plane.
RS identifies the eight edges of the Fano plane with the
\emph{eight-beat ledger cycle}:
every physical process completes exactly eight recognition
ticks~\cite{eight_phase_oracle_expanded}.
Specifically,

\begin{enumerate}\itemsep1pt
\item the real unit \(e_0\) acts during tick~0 (ledger initiation);
\item each imaginary $e_i$ rotates the recognition state
      during tick~\(i\pmod7\);
\item octonionic \emph{alternativity}
      \((xy)x = x(yx)\)
      guarantees cost conservation across the cycle.
\end{enumerate}

In the present paper the $e_i$ supply
\emph{braid coefficients} \(c_{ab}^{\;c}\in\{0,\pm1\}\)
coupling different $\mathrm{GL}(n)$ blocks.
Because the sum
\(\sum_{i=0}^{7}e_i = 0\),
the braiding operator adds no net quadratic cost,
ensuring the combined Hamiltonian
\(\mathbf H\) remains self-adjoint
(Section~\ref{sec:E8}).

Further categorical details appear in the
Ledger–Categorical manuscript~\cite{LedgerCategoricalFormulation};
only alternativity and the eight-beat sum rule are used below.

\subsection{Prime--Archimedean spectral measure}\label{sec:spectral-measure}

Fix a cuspidal automorphic representation
$\pi_{n}\subset L^{2}\!\bigl(\mathrm{GL}(n,\mathbb A_\mathbb Q)\bigr)$.
Let $\{\alpha_{1p},\dots,\alpha_{np}\}$ be its Satake parameters at
an unramified prime~$p$ (Section~\ref{sec:GLn-background}).
We define the \emph{log--spectral abscissæ}
\begin{equation}
  x_{p,i}\;:=\;\log p+\log|\alpha_{ip}|,
  \qquad
  1\le i\le n,
\end{equation}
and combine them with an Archimedean component to obtain the measure
\begin{equation}\label{eq:mu-n-def}
  d\mu_{n}(x)
  \;=\;
  \sum_{p}\;\sum_{i=1}^{n}\delta_{x=x_{p,i}}
  \;+\;
  x^{\,n-2}\,dx,
  \qquad x>0.
\end{equation}
The first term records arithmetic data of~$\pi_{n}$, while the
continuous term reproduces its $\Gamma$--factor
(cf.~\eqref{eq:L-completed}).

\subsection{The $\varphi$-weighted Hilbert space}\label{sec:Hn-space}

For the golden-ratio weight
$\rho(x)=e^{-2x/\varphi}$,
set
\begin{equation}\label{eq:Hn-def}
  \mathcal H_{n}
  \;=\;
  L^{2}\bigl(\mathbb R_{>0},\,\rho(x)\,d\mu_{n}(x)\bigr),
  \qquad
  \langle f,g\rangle_{n}
  =\!\int_{0}^{\infty}\!\!\overline{f(x)}\,g(x)\,
     e^{-2x/\varphi}\,d\mu_{n}(x).
\end{equation}
Define the diagonal multiplier
\begin{equation}\label{eq:Hn-operator}
   (H_{n}f)(x)\;:=\;x\,f(x),
   \qquad
   D(H_{n})
   =\Bigl\{f\in\mathcal H_{n}\ \Big|\ 
          \int x^{2}|f(x)|^{2}\rho(x)d\mu_{n}<\infty\Bigr\}.
\end{equation}

\subsection{Essential self‐adjointness}\label{sec:selfadjointness}

\begin{lemma}\label{lem:Hn-selfadjoint}
$H_{n}$ is essentially self-adjoint on the domain~\eqref{eq:Hn-operator}.
\end{lemma}

\begin{proof}[Proof sketch]
The Rankin--Selberg bound
$|\alpha_{ip}|\le p^{(n-1)/2-\theta}$ implies
$\mu_{n}([0,X])\le C e^{X}/X$.
Together with the exponential weight $\rho(x)$
this yields
\(\int_{X}^{\infty}x^{-2}\rho(x)d\mu_{n}(x)<\infty\),
so $x=0$ is in the limit--point case of Weyl's criterion;
the same holds at~$+\infty$.
Hence $H_{n}$ is essentially self-adjoint.
\end{proof}

\subsection{Hilbert–Schmidt property of the heat kernel}

\begin{lemma}\label{lem:HS}
For $A_{n}(s)=e^{-sH_{n}}$ one has
$A_{n}(s)\in\mathcal S_{2}(\mathcal H_{n})$
whenever $1/2<\Re s<1$.
\end{lemma}

\begin{proof}[Proof sketch]
Using~\eqref{eq:Hn-operator},
\[
  \|A_{n}(s)\|_{\mathcal S_{2}}^{2}
  =\int_{0}^{\infty}e^{-2(\Re s)x}\rho(x)\,d\mu_{n}(x)
  \;\le\;
  \int_{0}^{\infty}x^{-2}e^{-(2\Re s-1)x}\,dx
  <\infty.
\]
\end{proof}

\subsection{Fredholm determinant identity}\label{sec:Fredholm-identity}

\begin{theorem}[Golden‐ratio Fredholm identity on $\mathrm{GL}(n)$]\label{thm:fredholm}
Let $\pi_{n}$ be as above and set $A_{n}(s)=e^{-sH_{n}}$.
Then for $1/2<\Re s<1$
\begin{equation}\label{eq:fredholm-main}
  \det\nolimits_{2,\varphi}\!\bigl(I-A_{n}(s)\bigr)
  \;=\;
  \Lambda(s,\pi_{n})^{-1}.
\end{equation}
\end{theorem}

\begin{proof}[Idea of proof]
Expand
$\log\det_{2,\varphi}(I-A_{n}(s))
 =-\sum_{k\ge2}\tfrac1k\mathrm{tr}(A_{n}(s)^{k})$
and insert
$\mathrm{tr}\bigl(A_{n}(s)^{k}\bigr)
   =\sum_{p,i}(\alpha_{ip}^{-s}p^{-s})^{k}\log p$.
Regroup the absolutely convergent double sum to obtain
\(
\sum_{p,i}[\log(1-\alpha_{ip}p^{-s})+\alpha_{ip}p^{-s}].
\)
The linear counter‐term cancels precisely with the highest‐order pole
of the Archimedean factor thanks to the shift
$\varepsilon=\varphi^{-1}$,
leaving $\log\Lambda(s,\pi_{n})^{-1}$.
\end{proof}

\subsection{Critical‐line corollary}

\begin{corollary}[GRH for the $\mathrm{GL}(n)$ block]\label{cor:GRH-block}
All non‐trivial zeros of $\Lambda(s,\pi_{n})$ lie on $\Re s=\tfrac12$.
\end{corollary}

\begin{proof}
If $\Lambda(s,\pi_{n})=0$ in $1/2<\Re s<1$,
Eq.~\eqref{eq:fredholm-main} forces
$\det_{2,\varphi}(I-A_{n}(s))=0$,
so $A_{n}(s)$ has eigenvalue~$1$.
For the self‐adjoint $H_{n}$ this is possible
only when $|e^{-s\lambda}|=1$ for some $\lambda\in\mathbb R$,
i.e.\ $\Re s=\tfrac12$.
\end{proof}

\subsection{Octonions, Fano plane, and structure constants}
\label{sec:octonion-basics}

Write
\(
  \mathbb O=\mathrm{span}_{\mathbb R}\{e_{0},e_{1},\dots,e_{7}\}
\)
for the (real) octonion algebra with the \emph{Fano-plane}
multiplication table in Fig.~\ref{fig:fano}.
Every ordered triple $(i,j,k)$ lying on an oriented arrow
satisfies
\(e_{i}e_{j}=e_{k}=-e_{j}e_{i}\).
Introduce the \emph{structure constants}
\begin{equation}\label{eq:structure-constants}
   c_{ij}^{\;k}
   \;=\;
   \begin{cases}
      +1,& e_{i}e_{j}=+e_{k},\\
      -1,& e_{i}e_{j}=-e_{k},\\
      0 ,& \text{otherwise.}
   \end{cases}
\end{equation}
Two identities will be used below:

\begin{enumerate}\itemsep1pt
\item \emph{Alternativity}:\;
      $(xy)x = x(yx)$ for all $x,y\in\mathbb O$.
\item \emph{Eight-beat sum rule}:\;
      \(\displaystyle\sum_{i=0}^{7}e_{i}=0\).
\end{enumerate}

\subsection{The octonionic braid operator}\label{sec:braid-definition}

Combine the diagonal Hamiltonians
\(H_{n}\) of Section~\ref{sec:GLn} into
\begin{equation}\label{eq:H0}
   \mathbf H_{0}
   \;=\;
   \bigoplus_{n=1}^{8} H_{n},
   \qquad
   \mathcal H_{0}
   \;=\;
   \bigoplus_{n=1}^{8}\mathcal H_{n}.
\end{equation}
For $f\in\mathcal H_{0}$ write its components
\(f_{n}\in\mathcal H_{n}\).
Let $p,q$ run over primes
and $i,j$ index Satake parameters as before.
Define
\begin{equation}\label{eq:B-def}
   B\,e_{p,i}^{(m)}
   \;=\;
   \sum_{\substack{n\neq m\\ q,j}}
      \frac{c_{(m,i)(n,j)}^{\;\,r}}{\sqrt{\log p\,\log q}}\;
      e^{(n)}_{q,j},
\end{equation}
where $c_{ab}^{\;r}$ are the octonion constants~\eqref{eq:structure-constants}
with $r$ fixed to be the \emph{ledger axis}~$e_{0}$.
Because each $(m,i)$ interacts only with finitely many $(n,j)$
and the denominator grows like $\log p\,\log q$,
$B$ extends to a bounded operator
\(B:\mathcal H_{0}\to\mathcal H_{0}\)
that is \emph{Hilbert–Schmidt relative} to~$\mathbf H_{0}$.

\begin{lemma}\label{lem:B-HS-relative}
$B\bigl(\mathbf H_{0}+i\bigr)^{-1}\in\mathcal S_{2}(\mathcal H_{0})$.
\end{lemma}

\begin{proof}
For each basis pair $(p,i),(q,j)$
the matrix element is
\(\displaystyle
   \frac{c_{(m,i)(n,j)}^{\,r}}{\sqrt{\log p\log q}(\log q + i)},
\)
whose Hilbert–Schmidt norm
\(\sum_{p,q}\!\bigl(\log p\,\log q\bigr)^{-1}(\log q^{2}+1)^{-1}\)
converges by comparison with
\(\sum_{p}p^{-1-\delta}\) ($\delta>0$).
\end{proof}

\subsection{Self-adjointness of the braided operator}

\begin{proposition}\label{prop:H-braided-selfadjoint}
The operator
\(
   \displaystyle
   \mathbf H=\mathbf H_{0}+B
\)
is essentially self-adjoint on $D(\mathbf H_{0})$.
\end{proposition}

\begin{proof}
By Kato--Rellich,
if $B(\mathbf H_{0}+i)^{-1}\in\mathcal S_{2}$ (Lemma~\ref{lem:B-HS-relative})
then $B$ is $\mathbf H_{0}$–bounded with
relative bound $0$.
Hence $\mathbf H$ is self-adjoint on the domain of~$\mathbf H_{0}$.
\end{proof}

\subsection{Root-lattice realisation of \texorpdfstring{$E_{8}$}{E8}}
\label{sec:E8-spectrum}

Let
\(
   \Sigma
   =\bigl\{\log p+\log|\alpha_{ip}^{(n)}| + k\log\varphi
      \,\big|\,
      p\text{ prime},\,1\!\le\!i\!\le\!n,\,0\!\le\!n\!\le8,\,k\!\in\!\mathbb Z
      \bigr\}
\)
be the eigenvalue set of $\mathbf H_{0}$.
The braid operator $B$ adds \(\pm\log\varphi\)
along directions labelled by $e_{1},\dots,e_{7}$.
Because the octonion alternativity closes the
set of differences of eigenvalues under addition,
the quotient \(\Sigma/(\log\varphi)\mathbb Z\)
forms the $E_{8}$ root system
of $240$ length-squared–$2$ vectors.

\subsection{Block factorisation}\label{sec:block-factorisation}

Write
\(A_{n}(s)=e^{-sH_{n}}\),
\(A_{0}(s)=\bigoplus_{n=1}^{8}A_{n}(s)\),

\(K(s)=e^{-sB}\).
Because

\begin{equation}\label{eq:det-factorisation}
   \det_{2,\varphi}\!\bigl(I-e^{-s\mathbf H}\bigr)

      \det\nolimits_{2}\!\bigl(I - (I-A_{0}(s))^{-1}(K(s)-I)\bigr).
\end{equation}

\subsection{Eight–beat cancellation of the braid determinant}
\label{sec:braid-cancellation}

\begin{lemma}\label{lem:braid-det-one}
For every $s$ with $1/2<\Re s<1$,
\(
   \displaystyle
   \det\nolimits_{2}\!\bigl(I - (I-A_{0}(s))^{-1}(K(s)-I)\bigr)=1.
\)
\end{lemma}

\begin{proof}[Sketch]
Expand
$K(s)=\sum_{m\ge0}\frac{(-s)^{m}}{m!}B^{m}$.
Alternativity of $\mathbb O$ implies
$\sum_{i=0}^{7}B_{(i)}=0$
over one eight–beat cycle.
Consequently the diagonal of
$(I-A_{0}(s))^{-1}(K(s)-I)$
has vanishing trace.
Because the octonion structure constants satisfy
$c_{ab}^{\,\,r}=-c_{ba}^{\,\,r}$,
all eigenvalues occur in $\pm\lambda$ pairs,
so the determinant is~1.
\end{proof}

\subsection{The grand partition function}

\begin{definition}

\[
  Z_{E_{8}}(s)
  \;:=\;
  \prod_{n=1}^{8}\Lambda(s,\pi_{n}),

Combining Lemma~\ref{lem:braid-det-one}
with the factorisation~\eqref{eq:det-factorisation}
and the block identities~\eqref{eq:fredholm-main} yields

\begin{proposition}[Grand Fredholm identity]\label{prop:grand-fredholm}
For $1/2<\Re s<1$
\[
   \det_{2,\varphi}\!\bigl(I-e^{-s\mathbf H}\bigr)
   \;=\;
   Z_{E_{8}}(s)^{-1}.
\]
\end{proposition}

\subsection{Critical–line theorem}

\begin{theorem}\label{thm:grand-GRH}
All non–trivial zeros of $Z_{E_{8}}(s)$ satisfy
\(
   \Re s=\tfrac12.
\)
\end{theorem}

\begin{proof}
If $Z_{E_{8}}(s)=0$ in $1/2<\Re s<1$
then Proposition~\ref{prop:grand-fredholm}
gives $\det_{2,\varphi}(I-e^{-s\mathbf H})=0$,
so the self–adjoint operator $\mathbf H$
has an eigenvalue $\lambda$ with
$e^{-s\lambda}=1$.
This forces $|e^{-s\lambda}|=1$
and hence $\Re s=\tfrac12$.
\end{proof}

\subsection{Beyond \texorpdfstring{$\mathrm{GL}(8)$}{GL(8)}:
a glimpse toward full functoriality}
