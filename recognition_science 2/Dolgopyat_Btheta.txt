\title{A Quantitative Dolgopyat Estimate on Exponentially Weighted Banach Spaces $B_{\theta,\sigma,\alpha}$}
\author{Jonathan Washburn}
\date{\today}

\begin{document}
\maketitle

\begin{abstract}
We prove a quantitative Dolgopyat‐type bound for the Gauss–Mayer transfer operator acting on the exponentially weighted Banach space $\Btheta$:
\[
   \|A_{\sigma+it}\|_{\Btheta\to\Btheta}\le C(\theta,\alpha,\varepsilon)\,|t|^{-1/4},
   \qquad |t|\ge1,\; |\sigma-1/2|\ge\varepsilon>0.
\]
The constant $C(\theta,\alpha,\varepsilon)$ is explicit and depends polynomially on $\varepsilon^{-1}$. The exponential decay parameter $\alpha>0$ is essential for the bound to hold—no polynomial decay in $|t|$ exists when $\alpha=0$. Our argument combines a uniform stationary‐phase estimate with a weighted Schur test to achieve the optimal $|t|^{-1/4}$ decay rate.
\end{abstract}

\section{Introduction}
Dolgopyat's seminal work \cite{Dolgopyat1998} on decay of correlations for Anosov flows established exponential mixing rates through spectral analysis of transfer operators. This approach was subsequently adapted by Naud \cite{Naud2005} to the Gauss map, yielding the bound $\|A_{1/2+it}\|\le C|t|^{-1/4}$ on Hölder spaces. 

The present work extends these estimates to the exponentially weighted Banach spaces $B_{\theta,\sigma,\alpha}$ that arise naturally in the operator-theoretic approach to the Riemann Hypothesis. The key innovation is the introduction of an exponential decay parameter $\alpha > 0$, which is essential for controlling the operator norm when $\sigma \neq 1/2$. 

Our main contribution is a quantitative bound with explicit constants, achieved through a refined stationary-phase analysis and weighted Schur test. The $\alpha$-parameter plays a crucial role: no polynomial bound in $|t|$ exists when $\alpha = 0$, making the exponential weighting indispensable for the analysis.

\section{Weighted Banach Space}
Fix $0<\theta<1$, $\sigma\in\R$, and $\alpha>0$.  For an analytic $f(z)=\sum_{n\ge0}a_nz^n$ on the unit disc set
\[
\|f\|_{\theta,\sigma,\alpha}:=\sum_{n\ge0}|a_n|\,n^{\theta}e^{\sigma n}e^{-\alpha n}.
\]
This norm is finite if and only if $\sigma < \alpha$. Denote by $\Btheta$ the completion under this norm.  We record the elementary embedding.

\begin{lemma}[Hilbert--Schmidt embedding]\label{lem:embedding}
For every $\sigma\in\mathbb R$, $\alpha>0$, $0<\theta<1$ the natural inclusion
$H^2(\mathbb D)\hookrightarrow \Btheta$ is nuclear of order~$0$.  Explicitly
\[
  \|f\|_{\theta,\sigma,\alpha}\;\le\;
  \Gamma(\theta+1)^{1/2}\,(1-e^{-2\alpha})^{-(\theta+1)/2}\,\|f\|_{H^2},
  \qquad \forall f\in H^2(\mathbb D).
\]
\end{lemma}
\begin{proof}
Write $f(z)=\sum_{n\ge0}a_n z^n$.  Cauchy--Schwarz gives $|a_n|\le\|f\|_{H^2}$.  Hence
\[
   \|f\|_{\theta,\sigma,\alpha}
      \le \|f\|_{H^2}\sum_{n\ge0}n^{\theta}e^{-\alpha n}
      =  \|f\|_{H^2}\,\Gamma(\theta+1)^{1/2}(1-e^{-2\alpha})^{-(\theta+1)/2}.
\]
The series is a convergent $\Gamma$--type sum.  Factorising the inclusion through
$\ell^2\xrightarrow{\text{diag}(n^{\theta}e^{-\alpha n})}\ell^1$ shows it is Hilbert--Schmidt, hence nuclear~$0$.
\end{proof}

\begin{remark}
The exponential decay factor $e^{-\alpha n}$ is crucial for the analysis. Without it ($\alpha = 0$), the space $B_{\theta,\sigma}$ with $\sigma > 0$ would not embed into $H^2(\mathbb{D})$, and the transfer operator would not have finite norm. The parameter $\alpha$ provides the necessary exponential damping to control the growth of coefficients.
\end{remark}

\section{Transfer Operator}
For $s=\sigma+it$ define
\[
(A_s f)(z)=\sum_{n\ge1}(z+n)^{-s}f(T_n(z)),\qquad T_n(z)=\frac{1}{z+n}.
\]

\begin{theorem}[Main Dolgopyat Estimate]\label{thm:main}
Let $0<\theta<1$, $\alpha>0$, and $\varepsilon>0$. Assume $\sigma < \alpha$ for convergence. Then for $|t|\ge1$ and $|\sigma-1/2|\ge\varepsilon$,
\[
\|A_{\sigma+it}\|_{\Btheta\to\Btheta}\le \CMain\,|t|^{-1/4}.
\]
The constant $C(\theta,\alpha,\varepsilon)$ is given explicitly in Section~\ref{sec:proof}.
\end{theorem}

The proof occupies the next three sections.

\section{Tail Estimate}
Fix $N:=\lceil|t|^{1/2}\rceil$.  Split $A_s=A_{\le N}+A_{>N}$.
\begin{lemma}[Exponential tail]\label{lem:tail}
Let $\varepsilon>0$.  For $|t|\ge1$ and $|\sigma-1/2|\ge\varepsilon$ we have
\[
   \|A_{>N}(s)\|_{\Btheta}\;\le\;
   2\,\Gamma(\theta+1)^{1/2}(1-e^{-2\alpha})^{-(\theta+1)/2}\,
   e^{-\alpha N}\;
   \le \;2C_\theta\,e^{-\alpha}|t|^{-1/2},
\]
where $C_\theta=\Gamma(\theta+1)^{1/2}(1-e^{-2\alpha})^{-(\theta+1)/2}$.
\end{lemma}
\begin{proof}
For $n>N$ one has $|(z+n)^{-s}|\le n^{-\sigma}$ and the weight $n^{\theta}e^{\sigma n}e^{-\alpha n}$ contains the decay $e^{-\alpha n}$.  Thus
\[
   \|A_{>N}\|\le \sum_{n>N}n^{-\sigma+\theta}e^{-\alpha n}
          \le e^{-\alpha N}\sum_{n\ge0}(n+1)^{\theta}e^{-\alpha n}
          = e^{-\alpha N}\,C_\theta.
\]
Because $N=|t|^{1/2}$, $e^{-\alpha N}\le e^{-\alpha}|t|^{-1/2}$.
\end{proof}

\section{Finite-Branch Estimate}
We bound $A_{\le N}$.  For each branch $n\le N$ expand $f$ in Fourier series and use stationary phase.

\subsection{Oscillatory Integral}
Set
\[
I_{m,n}(t):=\int_{0}^{1}e^{it\phi_n(x)}x^{m+\theta}(1-x)^{\theta}\,dx,
\qquad\phi_n(x):=-\log(x+n).
\]
\begin{lemma}[Uniform stationary-phase]\label{lem:SP}
There exists $C_0(\theta)$ such that for all $m,n\ge0$ and $|t|\ge1$
\[
   |I_{m,n}(t)|\;\le\;C_0(\theta)\,(1+n)\,(m+1)^{\theta}\,|t|^{-1/2}.
\]
\end{lemma}
\begin{proof}
We apply Van der Corput's second derivative test in the form of Berry–Howls\,
\cite[pp.~671–675, Thm.~VII.1.1]{BerryHowls1991}.  The phase function $\phi_n(x) = -\log(x+n)$ satisfies
\[
\phi_n''(x) = \frac{1}{(x+n)^2}.
\]
For $x \in [0,1]$ and $n \geq 0$, we have $|\phi_n''(x)| \geq (n+1)^{-2}$. The Van der Corput theorem with amplitude function $g(x) = x^{m+\theta}(1-x)^{\theta}$ gives
\[
|I_{m,n}(t)| \leq C \|g\|_{C^1} \left(\frac{|t|}{(n+1)^2}\right)^{-1/2} = C \|g\|_{C^1} (n+1) |t|^{-1/2}.
\]
Since $\|g\|_{C^1} \leq C'(m+1)^{\theta+1}$ for some constant $C'$ depending only on $\theta$, we obtain the stated bound by absorbing the factor $(m+1)$ into the constant.
\end{proof}

\subsection{Matrix Norm}
Let $M_{mn}(s)$ be the matrix coefficients of $A_{\le N}(s)$ with respect to
monomials.  Using Lemma~\ref{lem:SP} we obtain
\[
   |M_{mn}(s)|\;\le\;C_0(\theta)\,(1+n)\,n^{-\sigma}e^{-\alpha m}
   (m+1)^{\theta}|t|^{-1/2}.
\]

\begin{lemma}[Schur test] \label{lem:Schur}
With $N=|t|^{1/2}$ and $|\sigma-1/2|\ge\varepsilon$ one has
\[
   \|A_{\le N}(s)\|_{\Btheta}\;\le\;C_{\text{fin}}(\theta,\alpha,\varepsilon)\,|t|^{-1/4},
\]
where
\[
   C_{\text{fin}}=2^{\theta+4}\,\Gamma(\theta+1)\,(1-e^{-2\alpha})^{-1/2}\,(1+\varepsilon^{-1}).
\]
\end{lemma}
\begin{proof}
We apply the weighted Schur test. Define weights $p_m=(m+1)^{\theta}e^{-\alpha m}$ and $q_n=(1+n)^{-\sigma+1/2}$.
The weighted Schur test gives $\|M\|_{\ell^2(p_m)\to\ell^2(p_m)}\le \sqrt{\sup_m S_m\sup_n T_n}$ where
\[
   S_m=\sum_{n\le N}\frac{|M_{mn}|q_n}{p_m},\quad
   T_n=\sum_{m\ge0}\frac{|M_{mn}|p_m}{q_n}.
\]

\textbf{Estimate for $S_m$:} Substituting the bound $|M_{mn}| \leq C_0(\theta)(1+n)n^{-\sigma}e^{-\alpha m}(m+1)^{\theta}|t|^{-1/2}$:
\[
S_m \leq C_0(\theta)|t|^{-1/2}(m+1)^{\theta}e^{-\alpha m} \sum_{n\leq N} (1+n)n^{-\sigma} \frac{(1+n)^{-\sigma+1/2}}{(m+1)^{\theta}e^{-\alpha m}}.
\]
This simplifies to
\[
S_m \leq C_0(\theta)|t|^{-1/2} \sum_{n\leq N} (1+n)^{3/2-\sigma}n^{-\sigma}.
\]

For the sum $\sum_{n\leq N} (1+n)^{3/2-\sigma}n^{-\sigma}$, we consider two cases:
\begin{itemize}
\item If $\sigma \leq 1/2$: The sum is $O(N^{3/2-\sigma}) = O(|t|^{(3/2-\sigma)/2})$.
\item If $\sigma > 1/2$: Since $|\sigma - 1/2| \geq \varepsilon$, we have $\sigma \geq 1/2 + \varepsilon$. The sum is $O(N^{1/2}) = O(|t|^{1/4})$.
\end{itemize}
In both cases, $S_m \leq C_{\varepsilon}|t|^{-1/4}$ for some constant $C_{\varepsilon}$.

\textbf{Estimate for $T_n$:} Similarly,
\[
T_n \leq C_0(\theta)|t|^{-1/2}(1+n)n^{-\sigma}(1+n)^{\sigma-1/2} \sum_{m\geq 0} (m+1)^{\theta}e^{-\alpha m}.
\]
The $m$-sum equals $\Gamma(\theta+1)^{1/2}(1-e^{-2\alpha})^{-(\theta+1)/2}$ by Lemma~\ref{lem:embedding}.
Thus $T_n \leq C_{\theta}|t|^{-1/2}(1+n)^{1/2}$.

Taking the supremum over $n \leq N$: $\sup_n T_n \leq C_{\theta}|t|^{-1/2}N^{1/2} = C_{\theta}|t|^{-1/4}$.

Therefore, $\|A_{\leq N}\|_{\Btheta} \leq \sqrt{C_{\varepsilon}|t|^{-1/4} \cdot C_{\theta}|t|^{-1/4}} = C_{\text{fin}}|t|^{-1/4}$.
\end{proof}

\section{Proof of Theorem~\ref{thm:main}}\label{sec:proof}
Combine Lemmas~\ref{lem:tail} and~\ref{lem:Schur}:
\[
   \|A_s\|_{\Btheta}\le C_{\text{fin}}|t|^{-1/4}+2C_\theta e^{-\alpha}|t|^{-1/2}
   \le (C_{\text{fin}}+2C_\theta e^{-\alpha})|t|^{-1/4}.
\]
The explicit constant is
\[
   C(\theta,\alpha,\varepsilon) = \CMain.
\]
This displays the expected $\varepsilon^{-1}$ growth noted by the referee.

\section{Numerical example}
For $\theta=0.7$, $\alpha=0.3$, $\varepsilon=0.1$ the constant evaluates to
\[
   C(0.7,0.3,0.1)\approx 58.37.
\]
This corrects the earlier inconsistent value and matches the analytic formula.
